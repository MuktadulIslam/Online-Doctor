/* First Create a database named 'onlinedoctor' */

CREATE TABLE `onlinedoctor`.`patients` (`username` VARCHAR(50) NOT NULL , `firstname` VARCHAR(20) NOT NULL , `lastname` VARCHAR(30) NULL DEFAULT NULL , `gender` VARCHAR(8) NOT NULL , `age` INT NOT NULL , `email` VARCHAR(100) NOT NULL , `fathername` VARCHAR(50) NULL DEFAULT NULL , `mothername` VARCHAR(50) NULL DEFAULT NULL , `password` VARCHAR(100) NOT NULL , `photo` VARCHAR(200) NOT NULL , `account_status` VARCHAR(10) NOT NULL , `deletion_date` DATE NULL DEFAULT NULL , PRIMARY KEY (`username`)) ENGINE = InnoDB;


CREATE TABLE `onlinedoctor`.`doctors` (`username` VARCHAR(50) NOT NULL , `firstname` VARCHAR(20) NOT NULL , `lastname` VARCHAR(30) NULL DEFAULT NULL , `gender` VARCHAR(8) NOT NULL , `age` INT NOT NULL , `email` VARCHAR(100) NOT NULL , `phonenumber` VARCHAR(15) NULL DEFAULT NULL , `fathername` VARCHAR(50) NULL DEFAULT NULL , `mothername` VARCHAR(50) NULL DEFAULT NULL , `doctorRegNumber` VARCHAR(20) NOT NULL , `specialization` VARCHAR(50) NOT NULL , `doctorDegree` VARCHAR(200) NOT NULL , `photo` VARCHAR(200) NULL DEFAULT NULL , `password` VARCHAR(50) NOT NULL , `account_status` VARCHAR(10) NOT NULL , `deletion_date` DATE NULL DEFAULT NULL , PRIMARY KEY (`username`)) ENGINE = InnoDB;

CREATE TABLE `onlinedoctor`.`appointments` (`appoinementID` VARCHAR(100) NOT NULL , `patientID` VARCHAR(50) NOT NULL , `doctorID` VARCHAR NOT NULL , `date` DATE NOT NULL , `reminder` BOOLEAN NULL , PRIMARY KEY (`appoinementID`)) ENGINE = InnoDB;

CREATE TABLE `onlinedoctor`.`admin` (`username` VARCHAR(30) NOT NULL , `phonenumber` VARCHAR(15) NULL DEFAULT NULL , `email` VARCHAR(100) NOT NULL , `password` VARCHAR(30) NOT NULL , PRIMARY KEY (`username`)) ENGINE = InnoDB;
